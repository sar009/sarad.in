def swap(i, j):
    """
    swap the variables
    :param i: index i
    :param j: index j
    :return: 
    """
    sqc[i], sqc[j] = sqc[j], sqc[i]


def heapify(end, i):
    """
    heap
    :param end: 
    :param i: 
    :return: 
    """
    l = 2 * i + 1
    r = 2 * (i + 1)
    max = i
    if l < end and sqc[i] < sqc[l]:
        max = l
    if r < end and sqc[max] < sqc[r]:
        max = r
    if max != i:
        swap(i, max)
        heapify(end, max)


def heap_sort():
    """
    start heap sort
    :return: 
    """
    end = len(sqc)
    start = end // 2 - 1 # use // instead of /
    for i in range(start, -1, -1):
        heapify(end, i)
    for i in range(end-1, 0, -1):
        swap(i, 0)
        heapify(i, 0)


sqc = [2, 7, 1, -2, 56, 5, 3]
